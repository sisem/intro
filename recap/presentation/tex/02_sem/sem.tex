  \frametitle{Prioritäten Inversion - Vererbung}

	\begin{itemize}
	  \item Tiefere priorität blockiert höhere Priorität
	\end{itemize}
	Lösung:
  \begin{itemize}
	  \item Prioräten Vererbung: "Lower-prio task inherits the prio of any higher-prio task pending on resource"
